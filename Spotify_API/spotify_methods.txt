# Tracks

## Get Audio features for a track(valence, danceability, energy, etc.)

sp.audio_features('7qiZfU4dY1lWllzX7mPBI3')

## Get popularity of a track
sp.track('7qiZfU4dY1lWllzX7mPBI3')['popularity']

## Get the audio analysis of a track
sp.audio_analysis('7qiZfU4dY1lWllzX7mPBI3')
